* Use of case in shell script for arithmetic opertaions

echo "Enter two number "
read a b
echo "1.add 2.sub 3.mult 4.div 5.exit"
read choice
case $choice in
        1) echo `expr $a + $b` ;;
        2) echo `expr $a - $b` ;;
        3) echo `expr $a \* $b` ;;
        4) echo `expr $a / $b` ;;
        5) echo "exit" ;;

* Use of case in shell script for string operations

a="abc"
b="efg"

if [ $a = $b ]
then
   echo "$a = $b : a is equal to b"
else
   echo "$a = $b: a is not equal to b"
fi

if [ $a != $b ]
then
   echo "$a != $b : a is not equal to b"
else
   echo "$a != $b: a is equal to b"
fi

if [ -z $a ]
then
   echo "-z $a : string length is zero"
else
   echo "-z $a : string length is not zero"
fi

if [ -n $a ]
then
   echo "-n $a : string length is not zero"
else
   echo "-n $a : string length is zero"
fi

if [ $a ]
then
   echo "$a : string is not empty"
else
   echo "$a : string is empty"
fi
